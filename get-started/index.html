# 快速上手

在项目中使用该 UI 库

import Vue from 'vue';
import { Button as ZButton } from 'zth';
import 'zth/dist/zth.css';

export default {
  components: { ZButton },
  name: 'app',
}

需要注意，样式文件需要单独引入
